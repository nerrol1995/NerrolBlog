JavaScript各种数据结构之间的转化方法

2018-05-16

1、嵌套对象数据格式转化单层对象数据格式

var showList = {
    "第一": {
      "11": "一年级",
      "12": "二年级",
      "13": "三年级",
      "14": "四年级",
      "15": "五年级",
      "16": "六年级"
    },
    "第二": {
      "2": "初一",
      "3": "初二",
      "4": "初三"
    },
    "第三": {
      "5": "高一",
      "6": "高二",
      "7": "高三"
    }
  
    //处理
    function formatObj (originObj) {
        var obj = {};
        for (let key in originObj) {
          for (let index in originObj[key]) {
            obj[index] = originObj[key][index]
          }
        }
        // console.log(obj);
        return obj;
      }
  console.log(obj)
    formatObj(showList);
